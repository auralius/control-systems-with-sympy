Define symbols¶

from sympy.physics.control.lti import TransferFunction
Kp, Ki, Kd = symbols('K_p K_i K_d', positive=True)

Ts      = symbols('T_s', positive=True)
t       = symbols('t', positive=True)
s       = symbols('s', complex=True)
z       = symbols('z')

e, u    = symbols('e u')
e0, e1, e2, u0, u1, u2 = symbols('e_k e_{k-1} e_{k-2} u_k u_{k-1} u_{k-2}')

PID control in s-domain¶

Let us start with defining our PID control in continuous form.

Gpid = Kp + Ki/s + Kd*s   # PID-control
mprint("G(s)= ", latex(Gpid))
\[\displaystyle G(s)= K_{d} s + \frac{K_{i}}{s} + K_{p}\]

From this point onwards, let us state for any arbitrary system:

  • \(e(t)\) is the input to the controller, which is the error between the target and the actual output of the system

  • \(u(t)\) is the computed control signal that will be sent to the system

  • \(T_s\) is the sampling period of the digital controller

PD-control with ZOH¶

Let us perform the following substitution:

\[ \frac{1}{s} \longleftarrow \frac{T_s}{1-z^{-1}} \]
Gpd = Kp + Kd*s   # PD-control
pid_eq1 = Eq(u, e*Gpd.subs(1/s, Ts / (1 - 1/z)))
mprint(latex(pid_eq1))
\[\displaystyle u = e \left(\frac{K_{d} \left(1 - \frac{1}{z}\right)}{T_{s}} + K_{p}\right)\]
pid_eq2 = factor(pid_eq1)
mprint(latex(pid_eq2))
\[\displaystyle u = \frac{e \left(K_{d} z - K_{d} + K_{p} T_{s} z\right)}{T_{s} z}\]
pid_eq3 = (Eq(numer(pid_eq2.rhs), pid_eq2.lhs * denom(pid_eq2.rhs)))
mprint(latex(pid_eq3))
\[\displaystyle e \left(K_{d} z - K_{d} + K_{p} T_{s} z\right) = T_{s} u z\]
pid_eq4 = expand(Eq(pid_eq3.lhs/z/Ts, pid_eq3.rhs/z/Ts ))
mprint(latex(pid_eq4))
\[\displaystyle \frac{K_{d} e}{T_{s}} - \frac{K_{d} e}{T_{s} z} + K_{p} e = u\]
pid_eq5 = (pid_eq4).subs(e/z**2, e2).subs(e/z, e1).subs(e, e0).subs(u/z, u1)
mprint(latex(pid_eq5))
\[\displaystyle \frac{K_{d} e_{k}}{T_{s}} - \frac{K_{d} e_{k-1}}{T_{s}} + K_{p} e_{k} = u\]
u_pd = simplify(collect(pid_eq5.lhs, [Kp, Kd]))
mprintb("u=", latex(u_pd))
\[\displaystyle \boxed{u=\frac{K_{d} \left(e_{k} - e_{k-1}\right) + K_{p} T_{s} e_{k}}{T_{s}}}\]

PID-control with ZOH¶

Gpid = Kp + Ki/s + Kd*s   # PID-control
pid_eq1 = Eq(u, e*Gpid.subs(1/s, Ts / (1 - 1/z)))
mprint(latex(pid_eq1))
\[\displaystyle u = e \left(\frac{K_{d} \left(1 - \frac{1}{z}\right)}{T_{s}} + \frac{K_{i} T_{s}}{1 - \frac{1}{z}} + K_{p}\right)\]
pid_eq2 = factor(pid_eq1)
mprint(latex(pid_eq2))
\[\displaystyle u = \frac{e \left(K_{d} z^{2} - 2 K_{d} z + K_{d} + K_{i} T_{s}^{2} z^{2} + K_{p} T_{s} z^{2} - K_{p} T_{s} z\right)}{T_{s} z \left(z - 1\right)}\]
pid_eq3 = (Eq(numer(pid_eq2.rhs), pid_eq2.lhs * denom(pid_eq2.rhs)))
mprint(latex(pid_eq3))
\[\displaystyle e \left(K_{d} z^{2} - 2 K_{d} z + K_{d} + K_{i} T_{s}^{2} z^{2} + K_{p} T_{s} z^{2} - K_{p} T_{s} z\right) = T_{s} u z \left(z - 1\right)\]
pid_eq4 = expand(Eq(pid_eq3.lhs/z**2/Ts, pid_eq3.rhs/z**2/Ts ))
mprint(latex(pid_eq4))
\[\displaystyle \frac{K_{d} e}{T_{s}} - \frac{2 K_{d} e}{T_{s} z} + \frac{K_{d} e}{T_{s} z^{2}} + K_{i} T_{s} e + K_{p} e - \frac{K_{p} e}{z} = u - \frac{u}{z}\]
pid_eq5 = expand(pid_eq4).subs(e/z**2, e2).subs(e/z, e1).subs(e, e0).subs(u/z, u1)
mprint(latex(pid_eq5))
\[\displaystyle \frac{K_{d} e_{k}}{T_{s}} - \frac{2 K_{d} e_{k-1}}{T_{s}} + \frac{K_{d} e_{k-2}}{T_{s}} + K_{i} T_{s} e_{k} + K_{p} e_{k} - K_{p} e_{k-1} = u - u_{k-1}\]
du_pid1 = ratsimp(pid_eq5.lhs)
mprint("u-u_{k-1}=", latex(du_pid1))
\[\displaystyle u-u_{k-1}=K_{i} T_{s} e_{k} + K_{p} e_{k} - K_{p} e_{k-1} + \frac{K_{d} e_{k} - 2 K_{d} e_{k-1} + K_{d} e_{k-2}}{T_{s}}\]
du_pid2 = collect(du_pid1, [Kp, Ki, Kd])
mprintb("u-u_{k-1}=", latex(du_pid2))
\[\displaystyle \boxed{u-u_{k-1}=\frac{K_{d} \left(e_{k} - 2 e_{k-1} + e_{k-2}\right)}{T_{s}} + K_{i} T_{s} e_{k} + K_{p} \left(e_{k} - e_{k-1}\right)}\]

PD control with bilinear transformation¶

Let us perform the following substitution:

\[ \frac{1}{s} \longleftarrow \frac{T_s}{2} \frac{z+1}{z-1} \]
Gpd = Kp + Kd*s   # PD-control
pid_eq1 = Eq(u, e*Gpd.subs(1/s, Ts/2 * (z+1)/(z-1) ) )
mprint(latex(pid_eq1))
\[\displaystyle u = e \left(\frac{2 K_{d} \left(z - 1\right)}{T_{s} \left(z + 1\right)} + K_{p}\right)\]
pid_eq2 = factor(pid_eq1)
mprint(latex(pid_eq2))
\[\displaystyle u = \frac{e \left(2 K_{d} z - 2 K_{d} + K_{p} T_{s} z + K_{p} T_{s}\right)}{T_{s} \left(z + 1\right)}\]
pid_eq3 = (Eq(numer(pid_eq2.rhs), pid_eq2.lhs * denom(pid_eq2.rhs)))
mprint(latex(pid_eq3))
\[\displaystyle e \left(2 K_{d} z - 2 K_{d} + K_{p} T_{s} z + K_{p} T_{s}\right) = T_{s} u \left(z + 1\right)\]
pid_eq4 = expand(Eq(pid_eq3.lhs/z/Ts, pid_eq3.rhs/z/Ts ))
mprint(latex(pid_eq4))
\[\displaystyle \frac{2 K_{d} e}{T_{s}} - \frac{2 K_{d} e}{T_{s} z} + K_{p} e + \frac{K_{p} e}{z} = u + \frac{u}{z}\]
pid_eq5 = expand(pid_eq4).subs(e/z**2, e2).subs(e/z, e1).subs(e, e0).subs(u/z, u1).subs(u,u0)
mprint(latex(pid_eq5))
\[\displaystyle \frac{2 K_{d} e_{k}}{T_{s}} - \frac{2 K_{d} e_{k-1}}{T_{s}} + K_{p} e_{k} + K_{p} e_{k-1} = u_{k} + u_{k-1}\]
du_pid1 = factor(pid_eq5.lhs)
mprint("u-u_{k-1}=", latex(du_pid1))
\[\displaystyle u-u_{k-1}=\frac{2 K_{d} e_{k} - 2 K_{d} e_{k-1} + K_{p} T_{s} e_{k} + K_{p} T_{s} e_{k-1}}{T_{s}}\]
du_pid2 = simplify(collect(du_pid1, [Kp, Ki, Kd]))
mprintb("u-u_{k-1}=", latex(du_pid2))
\[\displaystyle \boxed{u-u_{k-1}=\frac{2 K_{d} \left(e_{k} - e_{k-1}\right) + K_{p} T_{s} \left(e_{k} + e_{k-1}\right)}{T_{s}}}\]

PID control with bilinear transformation¶

Gpid = Kp + Ki/s + Kd*s   # PD-control
pid_eq1 = Eq(u, e*Gpid.subs(1/s, Ts/2 * (z+1)/(z-1) ) )
mprint(latex(pid_eq1))
\[\displaystyle u = e \left(\frac{2 K_{d} \left(z - 1\right)}{T_{s} \left(z + 1\right)} + \frac{K_{i} T_{s} \left(z + 1\right)}{2 \left(z - 1\right)} + K_{p}\right)\]
pid_eq2 = factor(pid_eq1)
mprint(latex(pid_eq2))
\[\displaystyle u = \frac{e \left(4 K_{d} z^{2} - 8 K_{d} z + 4 K_{d} + K_{i} T_{s}^{2} z^{2} + 2 K_{i} T_{s}^{2} z + K_{i} T_{s}^{2} + 2 K_{p} T_{s} z^{2} - 2 K_{p} T_{s}\right)}{2 T_{s} \left(z - 1\right) \left(z + 1\right)}\]
pid_eq3 = (Eq(numer(pid_eq2.rhs), pid_eq2.lhs * denom(pid_eq2.rhs)))
mprint(latex(pid_eq3))
\[\displaystyle e \left(4 K_{d} z^{2} - 8 K_{d} z + 4 K_{d} + K_{i} T_{s}^{2} z^{2} + 2 K_{i} T_{s}^{2} z + K_{i} T_{s}^{2} + 2 K_{p} T_{s} z^{2} - 2 K_{p} T_{s}\right) = 2 T_{s} u \left(z - 1\right) \left(z + 1\right)\]
pid_eq4 = expand(Eq(pid_eq3.lhs/z**2/2/Ts, pid_eq3.rhs/z**2/2/Ts ))
mprint(latex(pid_eq4))
\[\displaystyle \frac{2 K_{d} e}{T_{s}} - \frac{4 K_{d} e}{T_{s} z} + \frac{2 K_{d} e}{T_{s} z^{2}} + \frac{K_{i} T_{s} e}{2} + \frac{K_{i} T_{s} e}{z} + \frac{K_{i} T_{s} e}{2 z^{2}} + K_{p} e - \frac{K_{p} e}{z^{2}} = u - \frac{u}{z^{2}}\]
pid_eq5 = expand(pid_eq4).subs(e/z**2, e2).subs(e/z, e1).subs(e, e0).subs(u/z**2, u2).subs(u,u0)
mprint(latex(pid_eq5))
\[\displaystyle \frac{2 K_{d} e_{k}}{T_{s}} - \frac{4 K_{d} e_{k-1}}{T_{s}} + \frac{2 K_{d} e_{k-2}}{T_{s}} + \frac{K_{i} T_{s} e_{k}}{2} + K_{i} T_{s} e_{k-1} + \frac{K_{i} T_{s} e_{k-2}}{2} + K_{p} e_{k} - K_{p} e_{k-2} = u_{k} - u_{k-2}\]
du_pid1 = ratsimp(pid_eq5.lhs)
mprint("u-u_{k-2}=", latex(du_pid1))
\[\displaystyle u-u_{k-2}=\frac{K_{i} T_{s} e_{k}}{2} + K_{i} T_{s} e_{k-1} + \frac{K_{i} T_{s} e_{k-2}}{2} + K_{p} e_{k} - K_{p} e_{k-2} + \frac{2 K_{d} e_{k} - 4 K_{d} e_{k-1} + 2 K_{d} e_{k-2}}{T_{s}}\]
du_pid2 = collect(du_pid1, [Kp, Ki*Ts, 2*Kd])
mprintb("u-u_{k-2}=", latex(du_pid2))
\[\displaystyle \boxed{u-u_{k-2}=\frac{2 K_{d} \left(e_{k} - 2 e_{k-1} + e_{k-2}\right)}{T_{s}} + K_{i} T_{s} \left(\frac{e_{k}}{2} + e_{k-1} + \frac{e_{k-2}}{2}\right) + K_{p} \left(e_{k} - e_{k-2}\right)}\]
