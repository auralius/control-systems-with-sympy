Digital PID Control¶

Required imports¶

from IPython.core.display import HTML
import numpy as np
from sympy import *
from mathprint import *

init_printing(fontsize='5pt')

Define symbols¶

Kp, Ki, Kd = symbols('K_p K_i K_d', positive=True)

Ts, N     = symbols('T_s N', positive=True)
s, sd, si = symbols('s s_D s_I', complex=True)
z         = symbols('z')

e, u     = symbols('e u')
e0, e1, e2, u0, u1, u2 = symbols('e_k e_{k-1} e_{k-2} u_k u_{k-1} u_{k-2}')

PID control in s-domain¶

Let us start with defining our PID control in continuous form.

Gpid = Kp + Ki/s + Kd*s   # PID-control
mprint("G(s)= ", latex(Gpid))
\[\displaystyle G(s)= K_{d} s + \frac{K_{i}}{s} + K_{p}\]

From this point onwards, let us state for any arbitrary system:

  • \(e(t)\) is the input to the controller, which is the error between the target and the actual output of the system

  • \(u(t)\) is the computed control signal that will be sent to the system

  • \(T_s\) is the sampling period of the digital controller

PD-control with ZOH (Backward Euler)¶

Let us perform the following substitution:

\[ \frac{1}{s} \longleftarrow \frac{T_s}{1-z^{-1}} \]
Gpd = Kp + Kd*s   # PD-control
Gpd = Gpd.subs(1/s, Ts / (1 - 1/z))
mprint("G_{PD}=", latex(Gpd))

Gpd = collect(factor(Gpd),z)
mprintb("G_{PD}=", latex(Gpd))
\[\displaystyle G_{PD}=\frac{K_{d} \left(1 - \frac{1}{z}\right)}{T_{s}} + K_{p}\]
\[\displaystyle \boxed{G_{PD}=\frac{- K_{d} + z \left(K_{d} + K_{p} T_{s}\right)}{T_{s} z}}\]

Now, let us apply the input and compute the control output.

eq = Eq(u, Gpd * e)
mprint(latex(eq))

eq = (Eq(numer(eq.rhs), eq.lhs * denom(eq.rhs)))
mprint(latex(eq))

eq = Eq(eq.lhs/Ts/z, eq.rhs/Ts/z)
mprint(latex(eq))

eq = expand(eq).subs(e/z**2, e2).subs(e/z, e1).subs(e, e0).subs(u/z, u1)
mprint(latex(eq))
\[\displaystyle u = \frac{e \left(- K_{d} + z \left(K_{d} + K_{p} T_{s}\right)\right)}{T_{s} z}\]
\[\displaystyle e \left(- K_{d} + z \left(K_{d} + K_{p} T_{s}\right)\right) = T_{s} u z\]
\[\displaystyle \frac{e \left(- K_{d} + z \left(K_{d} + K_{p} T_{s}\right)\right)}{T_{s} z} = u\]
\[\displaystyle \frac{K_{d} e_{k}}{T_{s}} - \frac{K_{d} e_{k-1}}{T_{s}} + K_{p} e_{k} = u\]
upd = collect(eq.lhs, [Kp, Kd/Ts])
mprintb("u=", latex(upd))
\[\displaystyle \boxed{u=\frac{K_{d} \left(e_{k} - e_{k-1}\right)}{T_{s}} + K_{p} e_{k}}\]

PID-control with ZOH (Backward Euler)¶

Gpid = Kp + Ki/s + Kd*s   # PID-control
Gpid = Gpid.subs(1/s, Ts / (1 - 1/z))
mprint("G_{PID}=", latex(Gpid))

Gpid = collect(factor(Gpid), [z, z**2])
mprintb("G_{PID}=", latex(Gpid))
\[\displaystyle G_{PID}=\frac{K_{d} \left(1 - \frac{1}{z}\right)}{T_{s}} + \frac{K_{i} T_{s}}{1 - \frac{1}{z}} + K_{p}\]
\[\displaystyle \boxed{G_{PID}=\frac{K_{d} + z^{2} \left(K_{d} + K_{i} T_{s}^{2} + K_{p} T_{s}\right) + z \left(- 2 K_{d} - K_{p} T_{s}\right)}{T_{s} z \left(z - 1\right)}}\]

Now, let us apply the input and compute the control output.

eq = Eq(u, Gpid * e)
mprint(latex(eq))

eq = factor(eq)
mprint(latex(eq))

eq = Eq(numer(eq.rhs), eq.lhs * denom(eq.rhs))
mprint(latex(eq))

eq = expand(eq)
mprint(latex(eq))

eq = expand(Eq(eq.lhs/z**2/Ts, eq.rhs/z**2/Ts ))
mprint(latex(eq))

eq = expand(eq).subs(e/z**2, e2).subs(e/z, e1).subs(e, e0).subs(u/z, u1)
mprint(latex(eq))

eq = ratsimp(eq.lhs)
mprint("u-u_{k-1}=", latex(eq))

eq = collect(eq, [Kp, Ki, Kd])
mprintb("u-u_{k-1}=", latex(eq))
\[\displaystyle u = \frac{e \left(K_{d} + z^{2} \left(K_{d} + K_{i} T_{s}^{2} + K_{p} T_{s}\right) + z \left(- 2 K_{d} - K_{p} T_{s}\right)\right)}{T_{s} z \left(z - 1\right)}\]
\[\displaystyle u = \frac{e \left(K_{d} z^{2} - 2 K_{d} z + K_{d} + K_{i} T_{s}^{2} z^{2} + K_{p} T_{s} z^{2} - K_{p} T_{s} z\right)}{T_{s} z \left(z - 1\right)}\]
\[\displaystyle e \left(K_{d} z^{2} - 2 K_{d} z + K_{d} + K_{i} T_{s}^{2} z^{2} + K_{p} T_{s} z^{2} - K_{p} T_{s} z\right) = T_{s} u z \left(z - 1\right)\]
\[\displaystyle K_{d} e z^{2} - 2 K_{d} e z + K_{d} e + K_{i} T_{s}^{2} e z^{2} + K_{p} T_{s} e z^{2} - K_{p} T_{s} e z = T_{s} u z^{2} - T_{s} u z\]
\[\displaystyle \frac{K_{d} e}{T_{s}} - \frac{2 K_{d} e}{T_{s} z} + \frac{K_{d} e}{T_{s} z^{2}} + K_{i} T_{s} e + K_{p} e - \frac{K_{p} e}{z} = u - \frac{u}{z}\]
\[\displaystyle \frac{K_{d} e_{k}}{T_{s}} - \frac{2 K_{d} e_{k-1}}{T_{s}} + \frac{K_{d} e_{k-2}}{T_{s}} + K_{i} T_{s} e_{k} + K_{p} e_{k} - K_{p} e_{k-1} = u - u_{k-1}\]
\[\displaystyle u-u_{k-1}=K_{i} T_{s} e_{k} + K_{p} e_{k} - K_{p} e_{k-1} + \frac{K_{d} e_{k} - 2 K_{d} e_{k-1} + K_{d} e_{k-2}}{T_{s}}\]
\[\displaystyle \boxed{u-u_{k-1}=\frac{K_{d} \left(e_{k} - 2 e_{k-1} + e_{k-2}\right)}{T_{s}} + K_{i} T_{s} e_{k} + K_{p} \left(e_{k} - e_{k-1}\right)}\]

PD-control with bilinear transformation¶

Let us perform the following substitution:

\[ \frac{1}{s} \longleftarrow \frac{T_s}{2} \frac{z+1}{z-1} \]
Gpd = Kp + Kd*s   # PD-control
Gpd = Gpd.subs(1/s, Ts/2 * (z+1)/(z-1) )
mprint("G_{PD}=", latex(Gpd))

Gpd = collect(factor(Gpd),z)
mprintb("G_{PD}=", latex(Gpd))
\[\displaystyle G_{PD}=\frac{2 K_{d} \left(z - 1\right)}{T_{s} \left(z + 1\right)} + K_{p}\]
\[\displaystyle \boxed{G_{PD}=\frac{- 2 K_{d} + K_{p} T_{s} + z \left(2 K_{d} + K_{p} T_{s}\right)}{T_{s} \left(z + 1\right)}}\]
eq = Eq(u, Gpd * e)
mprint(latex(eq))

eq = factor(eq)
mprint(latex(eq))

eq = Eq(numer(eq.rhs), eq.lhs * denom(eq.rhs))
mprint(latex(eq))

eq = expand(eq)
mprint(latex(eq))

eq = expand(Eq(eq.lhs/z**2/Ts, eq.rhs/z**2/Ts ))
mprint(latex(eq))

eq = expand(eq).subs(e/z**2, e2).subs(e/z, e1).subs(e, e0).subs(u/z**2, u2).subs(u/z, u1)
mprint(latex(eq))

eq = ratsimp(eq.lhs)
mprint("u-u_{k-1}=", latex(eq))

eq = collect(eq, [Kp, Ki, 2*Kd])
mprintb("u-u_{k-1}=", latex(eq))
\[\displaystyle u = \frac{e \left(- 2 K_{d} + K_{p} T_{s} + z \left(2 K_{d} + K_{p} T_{s}\right)\right)}{T_{s} \left(z + 1\right)}\]
\[\displaystyle u = \frac{e \left(2 K_{d} z - 2 K_{d} + K_{p} T_{s} z + K_{p} T_{s}\right)}{T_{s} \left(z + 1\right)}\]
\[\displaystyle e \left(2 K_{d} z - 2 K_{d} + K_{p} T_{s} z + K_{p} T_{s}\right) = T_{s} u \left(z + 1\right)\]
\[\displaystyle 2 K_{d} e z - 2 K_{d} e + K_{p} T_{s} e z + K_{p} T_{s} e = T_{s} u z + T_{s} u\]
\[\displaystyle \frac{2 K_{d} e}{T_{s} z} - \frac{2 K_{d} e}{T_{s} z^{2}} + \frac{K_{p} e}{z} + \frac{K_{p} e}{z^{2}} = \frac{u}{z} + \frac{u}{z^{2}}\]
\[\displaystyle \frac{2 K_{d} e_{k-1}}{T_{s}} - \frac{2 K_{d} e_{k-2}}{T_{s}} + K_{p} e_{k-1} + K_{p} e_{k-2} = u_{k-1} + u_{k-2}\]
\[\displaystyle u-u_{k-1}=K_{p} e_{k-1} + K_{p} e_{k-2} + \frac{2 K_{d} e_{k-1} - 2 K_{d} e_{k-2}}{T_{s}}\]
\[\displaystyle \boxed{u-u_{k-1}=\frac{2 K_{d} \left(e_{k-1} - e_{k-2}\right)}{T_{s}} + K_{p} \left(e_{k-1} + e_{k-2}\right)}\]

PID-control with bilinear transformation¶

Gpid = Kp + Ki/s + Kd*s   # PID-control
Gpid = Gpid.subs(1/s, Ts/2 * (z+1)/(z-1) )
mprint("G_{PID}=", latex(Gpid))

Gpid = factor(Gpid)

Gpid = collect(numer(Gpid), [z, z**2]) /  expand(denom(Gpid))
mprintb("G_{PID}=", latex(Gpid))
\[\displaystyle G_{PID}=\frac{2 K_{d} \left(z - 1\right)}{T_{s} \left(z + 1\right)} + \frac{K_{i} T_{s} \left(z + 1\right)}{2 \left(z - 1\right)} + K_{p}\]
\[\displaystyle \boxed{G_{PID}=\frac{4 K_{d} + K_{i} T_{s}^{2} - 2 K_{p} T_{s} + z^{2} \left(4 K_{d} + K_{i} T_{s}^{2} + 2 K_{p} T_{s}\right) + z \left(- 8 K_{d} + 2 K_{i} T_{s}^{2}\right)}{2 T_{s} z^{2} - 2 T_{s}}}\]

Now, let us apply the input to the output.

eq = Eq(u, Gpid * e)
mprint(latex(eq))

eq = factor(eq)
mprint(latex(eq))

eq = Eq(numer(eq.rhs), eq.lhs * denom(eq.rhs))
mprint(latex(eq))

eq = expand(eq)
mprint(latex(eq))

eq = expand(Eq(eq.lhs/z**2/Ts, eq.rhs/z**2/Ts ))
mprint(latex(eq))

eq = expand(eq).subs(e/z**2, e2).subs(e/z, e1).subs(e, e0).subs(u/z**2, u2)
mprint(latex(eq))

eq = ratsimp(eq.lhs)
mprint("u-u_{k-2}=", latex(eq))

eq_ = collect(eq, [2*Kp, Ki*Ts, 4*Kd])
mprintb("u-u_{k-2}=", latex(eq_))

eq_ = collect(expand(eq), [e0, e1, e2])
mprintb("u-u_{k-2}=", latex(eq_))
\[\displaystyle u = \frac{e \left(4 K_{d} + K_{i} T_{s}^{2} - 2 K_{p} T_{s} + z^{2} \left(4 K_{d} + K_{i} T_{s}^{2} + 2 K_{p} T_{s}\right) + z \left(- 8 K_{d} + 2 K_{i} T_{s}^{2}\right)\right)}{2 T_{s} z^{2} - 2 T_{s}}\]
\[\displaystyle u = \frac{e \left(4 K_{d} z^{2} - 8 K_{d} z + 4 K_{d} + K_{i} T_{s}^{2} z^{2} + 2 K_{i} T_{s}^{2} z + K_{i} T_{s}^{2} + 2 K_{p} T_{s} z^{2} - 2 K_{p} T_{s}\right)}{2 T_{s} \left(z - 1\right) \left(z + 1\right)}\]
\[\displaystyle e \left(4 K_{d} z^{2} - 8 K_{d} z + 4 K_{d} + K_{i} T_{s}^{2} z^{2} + 2 K_{i} T_{s}^{2} z + K_{i} T_{s}^{2} + 2 K_{p} T_{s} z^{2} - 2 K_{p} T_{s}\right) = 2 T_{s} u \left(z - 1\right) \left(z + 1\right)\]
\[\displaystyle 4 K_{d} e z^{2} - 8 K_{d} e z + 4 K_{d} e + K_{i} T_{s}^{2} e z^{2} + 2 K_{i} T_{s}^{2} e z + K_{i} T_{s}^{2} e + 2 K_{p} T_{s} e z^{2} - 2 K_{p} T_{s} e = 2 T_{s} u z^{2} - 2 T_{s} u\]
\[\displaystyle \frac{4 K_{d} e}{T_{s}} - \frac{8 K_{d} e}{T_{s} z} + \frac{4 K_{d} e}{T_{s} z^{2}} + K_{i} T_{s} e + \frac{2 K_{i} T_{s} e}{z} + \frac{K_{i} T_{s} e}{z^{2}} + 2 K_{p} e - \frac{2 K_{p} e}{z^{2}} = 2 u - \frac{2 u}{z^{2}}\]
\[\displaystyle \frac{4 K_{d} e_{k}}{T_{s}} - \frac{8 K_{d} e_{k-1}}{T_{s}} + \frac{4 K_{d} e_{k-2}}{T_{s}} + K_{i} T_{s} e_{k} + 2 K_{i} T_{s} e_{k-1} + K_{i} T_{s} e_{k-2} + 2 K_{p} e_{k} - 2 K_{p} e_{k-2} = 2 u - 2 u_{k-2}\]
\[\displaystyle u-u_{k-2}=K_{i} T_{s} e_{k} + 2 K_{i} T_{s} e_{k-1} + K_{i} T_{s} e_{k-2} + 2 K_{p} e_{k} - 2 K_{p} e_{k-2} + \frac{4 K_{d} e_{k} - 8 K_{d} e_{k-1} + 4 K_{d} e_{k-2}}{T_{s}}\]
\[\displaystyle \boxed{u-u_{k-2}=\frac{4 K_{d} \left(e_{k} - 2 e_{k-1} + e_{k-2}\right)}{T_{s}} + K_{i} T_{s} \left(e_{k} + 2 e_{k-1} + e_{k-2}\right) + 2 K_{p} \left(e_{k} - e_{k-2}\right)}\]
\[\displaystyle \boxed{u-u_{k-2}=e_{k} \left(\frac{4 K_{d}}{T_{s}} + K_{i} T_{s} + 2 K_{p}\right) + e_{k-1} \left(- \frac{8 K_{d}}{T_{s}} + 2 K_{i} T_{s}\right) + e_{k-2} \left(\frac{4 K_{d}}{T_{s}} + K_{i} T_{s} - 2 K_{p}\right)}\]

PID-control with ZOH (both Forward and Backward Euler) and derivative filter¶

If we use MATLAB Simulink, we will notice that the MATLAB Simulink allows us to use various discrete forms for its PID control. Additionally, in MATLAB Simulink, the derivative term is equpped with a first-order low-pass filter.

For validation purposes, let us use the default options where MATLAB Simulink presents us with the following discrete PID formula:

\[ K_p + K_i \cdot T_s \frac{1}{z-1}+ K_d \frac{N}{1+N \cdot T_s \frac{1}{z-1}} \]

We can see that:

  • for the integral term: ZOH with Forward Euler:

\[ \frac{1}{s_D} \longleftarrow \frac{T_s}{z-1} \]

  • for the derivative term: ZOH with backward Euler:

\[ s_I \longleftarrow \frac{1}{T_s}\frac{z-1}{z} \]

Since the two \(s\) variables are going to be substituted differently, we introduce \(s_I\) for the integral term and \(s_D\) for the derivative term.

Gpid = Kp + Ki/si + Kd*N/(1+N/sd)   # PID-control
mprint("G_{PID}=", latex(Gpid))

Gpid = Gpid.subs(([1/si, Ts / (z-1)], 
                  [sd  , (z-1) / (Ts *z)]))
mprint("G_{PID}=", latex(Gpid))
\[\displaystyle G_{PID}=\frac{K_{d} N}{\frac{N}{s_{D}} + 1} + \frac{K_{i}}{s_{I}} + K_{p}\]
\[\displaystyle G_{PID}=\frac{K_{d} N}{\frac{N T_{s} z}{z - 1} + 1} + \frac{K_{i} T_{s}}{z - 1} + K_{p}\]
eq = Eq(u, Gpid * e)
mprint(latex(eq))

eq = factor(eq)
mprint(latex(eq))

eq = Eq(numer(eq.rhs), eq.lhs * denom(eq.rhs))
mprint(latex(eq))

#eq = expand(eq)
#mprint(latex(eq))

eq = expand(Eq(eq.lhs/z**2/Ts/N, eq.rhs/z**2/Ts/N ))
mprint(latex(eq))

eq = (eq.subs(e/z**2, e2).subs(e/z, e1).subs(e, e0).subs(u/z**2, u2).subs(u/z, u1).subs(u, u0))
mprint("\\small ", latex(eq))

eq =Eq(collect(eq.rhs, [u0, u1, u2]), collect(eq.lhs, [e0, e1, e2]))
mprintb("\\small ", latex(eq))
\[\displaystyle u = e \left(\frac{K_{d} N}{\frac{N T_{s} z}{z - 1} + 1} + \frac{K_{i} T_{s}}{z - 1} + K_{p}\right)\]
\[\displaystyle u = \frac{e \left(K_{d} N z^{2} - 2 K_{d} N z + K_{d} N + K_{i} N T_{s}^{2} z + K_{i} T_{s} z - K_{i} T_{s} + K_{p} N T_{s} z^{2} - K_{p} N T_{s} z + K_{p} z^{2} - 2 K_{p} z + K_{p}\right)}{\left(z - 1\right) \left(N T_{s} z + z - 1\right)}\]
\[\displaystyle e \left(K_{d} N z^{2} - 2 K_{d} N z + K_{d} N + K_{i} N T_{s}^{2} z + K_{i} T_{s} z - K_{i} T_{s} + K_{p} N T_{s} z^{2} - K_{p} N T_{s} z + K_{p} z^{2} - 2 K_{p} z + K_{p}\right) = u \left(z - 1\right) \left(N T_{s} z + z - 1\right)\]
\[\displaystyle \frac{K_{d} e}{T_{s}} - \frac{2 K_{d} e}{T_{s} z} + \frac{K_{d} e}{T_{s} z^{2}} + \frac{K_{i} T_{s} e}{z} + \frac{K_{i} e}{N z} - \frac{K_{i} e}{N z^{2}} + K_{p} e - \frac{K_{p} e}{z} + \frac{K_{p} e}{N T_{s}} - \frac{2 K_{p} e}{N T_{s} z} + \frac{K_{p} e}{N T_{s} z^{2}} = u - \frac{u}{z} + \frac{u}{N T_{s}} - \frac{2 u}{N T_{s} z} + \frac{u}{N T_{s} z^{2}}\]
\[\displaystyle \small \frac{K_{d} e_{k}}{T_{s}} - \frac{2 K_{d} e_{k-1}}{T_{s}} + \frac{K_{d} e_{k-2}}{T_{s}} + K_{i} T_{s} e_{k-1} + \frac{K_{i} e_{k-1}}{N} - \frac{K_{i} e_{k-2}}{N} + K_{p} e_{k} - K_{p} e_{k-1} + \frac{K_{p} e_{k}}{N T_{s}} - \frac{2 K_{p} e_{k-1}}{N T_{s}} + \frac{K_{p} e_{k-2}}{N T_{s}} = u_{k} - u_{k-1} + \frac{u_{k}}{N T_{s}} - \frac{2 u_{k-1}}{N T_{s}} + \frac{u_{k-2}}{N T_{s}}\]
\[\displaystyle \boxed{\small u_{k} \left(1 + \frac{1}{N T_{s}}\right) + u_{k-1} \left(-1 - \frac{2}{N T_{s}}\right) + \frac{u_{k-2}}{N T_{s}} = e_{k} \left(\frac{K_{d}}{T_{s}} + K_{p} + \frac{K_{p}}{N T_{s}}\right) + e_{k-1} \left(- \frac{2 K_{d}}{T_{s}} + K_{i} T_{s} + \frac{K_{i}}{N} - K_{p} - \frac{2 K_{p}}{N T_{s}}\right) + e_{k-2} \left(\frac{K_{d}}{T_{s}} - \frac{K_{i}}{N} + \frac{K_{p}}{N T_{s}}\right)}\]
