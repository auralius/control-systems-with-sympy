System and the control equations#

The diagram above corresponds to a unity-feedback loop with

\[ G(s)=K_p+\frac{K_i}{s}+K_d \]

and plant

\[ H(s)=\frac{1}{\tau s+1}. \]

Thus, the closed-loop transfer function is

\[ \frac{C(s)}{R(s)} = \frac{\left(K_p+\frac{K_i}{s}+K_ds\right)\frac{1}{\tau s+1}} {1+\left(K_p+\frac{K_i}{s}+K_ds\right)\frac{1}{\tau s+1}} = \frac{K_d s^2+K_p s+K_i} {(\tau+K_d)s^2+(1+K_p)s+K_i} \]
H = 1 / (tau*s+1)
G = Kp + Ki/s + Kd*s   # PID-control
Q = factor(((G*H / (1 + G*H))))

print("first-order system:")
mprint("H(s)=", latex(H))

print("PID-control:")
mprint("G(s)= ", latex(G))

print("the resulting closed-loop system:")
mprint("Q(s)=", latex(Q))

first-order system:
\[\displaystyle H(s)=\frac{1}{s \tau + 1}\]
PID-control:
\[\displaystyle G(s)= K_{d} s + \frac{K_{i}}{s} + K_{p}\]

\[\displaystyle Q(s)=\frac{K_{d} s^{2} + K_{i} + K_{p} s}{K_{d} s^{2} + K_{i} + K_{p} s + s^{2} \tau + s}\]

Interactive PID Response of a First-Order Plant#

This section demonstrates the step response of a first-order plant controlled using a PID controller. The goal is to observe how the controller gains \(K_p\), \(K_i\), and \(K_d\) affect the closed-loop response.

The plant is modeled as a standard first-order system with a time constant of \(\tau\).

The notebook then plots the response \(c(t)\) to a unit-step input.

The notebook also calculates common time-domain response characteristics, including:

  • rise time,

  • peak time,

  • maximum overshoot,

  • settling time using the 5% band,

  • settling time using the 2% band,

  • closed-loop pole locations.

# Interactive PID / PI / PD response of a first-order plant
# This version uses derived closed-form equations only.
#
# Plant:
#     H(s) = 1 / (tau*s + 1)
#
# Controller:
#     G(s) = Kp + Ki/s + Kd*s
#
# Cases:
#     Ki > 0 : PID / PI case, second-order closed-loop response
#     Ki = 0 : PD / P case, first-order closed-loop response
#
# PI is obtained by setting Kd = 0 and Ki > 0.
# PD is obtained by setting Ki = 0 and Kd > 0.
# P  is obtained by setting Ki = 0 and Kd = 0.

import numpy as np
import matplotlib.pyplot as plt
import ipywidgets as widgets
from IPython.display import display
from scipy.optimize import brentq

# Safer for Binder/Colab because it avoids calling a system LaTeX executable.
plt.rcParams["text.usetex"] = False


def closed_loop_denominator_pid_fo(Kp, Ki, Kd, tau):
    """
    PID/PI closed-loop denominator for the first-order plant:

        (tau + Kd)*s^2 + (1 + Kp)*s + Ki
    """
    return [
        tau + Kd,
        1.0 + Kp,
        Ki
    ]


def closed_loop_denominator_pd_fo(Kp, Kd, tau):
    """
    PD/P closed-loop denominator for the first-order plant:

        (tau + Kd)*s + (1 + Kp)
    """
    return [
        tau + Kd,
        1.0 + Kp
    ]


def closed_loop_poles_and_a_pid_fo(Kp, Ki, Kd, tau):
    """
    Compute PID/PI closed-loop poles p_i, then define a_i = -p_i.
    """
    den = closed_loop_denominator_pid_fo(Kp, Ki, Kd, tau)
    poles = np.roots(den)
    a = -poles
    return poles, a, den


def closed_loop_pole_b_pd_fo(Kp, Kd, tau):
    """
    Compute the PD/P closed-loop pole p = -b.
    """
    den = closed_loop_denominator_pd_fo(Kp, Kd, tau)
    b = (1.0 + Kp) / (tau + Kd)
    pole = -b
    return np.array([pole]), b, den


def closed_form_pid_fo_response(t, Kp, Ki, Kd, tau, a):
    """
    Derived PID/PI closed-form unit-step response for the first-order plant.

    For Ki > 0:

        C(s) =
            (Kd*s^2 + Kp*s + Ki)
            -------------------------------------------------
            s*((tau+Kd)*s^2 + (1+Kp)*s + Ki)

    If:

        ((tau+Kd)*s^2 + (1+Kp)*s + Ki)
        =
        (tau+Kd)*(s+a1)*(s+a2)

    then:

        c(t)
        =
        Ki/((tau+Kd)*a1*a2)

        + (Kd*a1^2 - Kp*a1 + Ki)
          / ((tau+Kd)*a1*(a1-a2)) * exp(-a1*t)

        + (Kd*a2^2 - Kp*a2 + Ki)
          / ((tau+Kd)*a2*(a2-a1)) * exp(-a2*t)

    This assumes a1 and a2 are distinct.
    """
    t = np.asarray(t, dtype=float)

    A = tau + Kd
    a1, a2 = a

    c_inf = Ki / (A * a1 * a2)

    term_a1 = ((Kd*a1**2 - Kp*a1 + Ki) * np.exp(-a1*t)) / (
        A * a1 * (a1 - a2)
    )

    term_a2 = ((Kd*a2**2 - Kp*a2 + Ki) * np.exp(-a2*t)) / (
        A * a2 * (a2 - a1)
    )

    c = c_inf + term_a1 + term_a2

    return np.real_if_close(c, tol=1000)


def closed_form_pd_fo_response(t, Kp, Kd, tau, b):
    """
    Derived PD/P closed-form unit-step response for the first-order plant.

    For Ki = 0:

        C(s) =
            (Kd*s + Kp)
            ------------------------------
            s*((tau+Kd)*s + (1+Kp))

    Let:

        b = (1 + Kp)/(tau + Kd)

    Then:

        c(t)
        =
        Kp/((tau+Kd)*b)
        +
        (Kd*b - Kp)/((tau+Kd)*b) * exp(-b*t)
    """
    t = np.asarray(t, dtype=float)

    A = tau + Kd

    c_inf = Kp / (A * b)
    transient = ((Kd*b - Kp) * np.exp(-b*t)) / (A * b)

    c = c_inf + transient

    return np.real_if_close(c, tol=1000)


def first_crossing_time(func, t_grid, y_grid, level):
    """Find first time where y(t) crosses a specified level."""
    f_grid = y_grid - level

    for k in range(len(t_grid) - 1):
        if f_grid[k] == 0:
            return t_grid[k]

        if f_grid[k] * f_grid[k + 1] < 0:
            t_left = t_grid[k]
            t_right = t_grid[k + 1]

            try:
                return brentq(
                    lambda tau_: func(tau_) - level,
                    t_left,
                    t_right
                )
            except ValueError:
                return None

    return None


def settling_time(t, y, y_final, tol):
    """Settling time: first time after which response stays inside ±tol band."""
    lower = (1.0 - tol) * y_final
    upper = (1.0 + tol) * y_final

    inside = (y >= lower) & (y <= upper)

    for k in range(len(t)):
        if np.all(inside[k:]):
            return t[k]

    return None


def roots_are_distinct(values, tol=1e-7):
    """Check whether root constants are sufficiently distinct."""
    values = np.asarray(values)

    for i in range(len(values)):
        for j in range(i + 1, len(values)):
            if abs(values[i] - values[j]) <= tol:
                return False

    return True


def plot_pid_fo_response(Kp=10.0, Ki=3.0, Kd=1.0,
                         tau=1.0, t_final=20.0):

    # Basic validation
    if tau <= 0:
        print("Please use tau > 0.")
        return

    if Kp < 0:
        print("Please use Kp >= 0.")
        return

    if Ki < 0:
        print("Please use Ki >= 0.")
        return

    if Kd < 0:
        print("Please use Kd >= 0.")
        return

    if tau + Kd <= 0:
        print("Please use tau + Kd > 0.")
        return

    if t_final <= 0:
        print("Please use t final > 0.")
        return

    t = np.linspace(0, t_final, 2500)

    # ------------------------------------------------------------
    # Case 1: PID / PI
    # ------------------------------------------------------------
    if Ki > 0:
        controller_type = "PID" if Kd > 0 else "PI"

        poles, a, den = closed_loop_poles_and_a_pid_fo(Kp, Ki, Kd, tau)
        stable = np.all(np.real(poles) < 0)

        if not roots_are_distinct(a):
            print("The closed-loop poles are repeated or too close.")
            print("The derived PID/PI partial-fraction equation assumes distinct poles.")
            print("Try changing Kp, Ki, Kd, or tau slightly.")
            return

        c_complex = closed_form_pid_fo_response(t, Kp, Ki, Kd, tau, a)
        c = np.real(c_complex)

        c_inf_from_poles = Ki / ((tau + Kd) * np.prod(a))
        c_inf = float(np.real_if_close(c_inf_from_poles))

        def c_of_t(tau_):
            y = closed_form_pid_fo_response(
                np.array([tau_]), Kp, Ki, Kd, tau, a
            )
            return float(np.real(y[0]))

        pole_constants_name = "a_i = -p_i"

    # ------------------------------------------------------------
    # Case 2: PD / P
    # ------------------------------------------------------------
    else:
        controller_type = "PD" if Kd > 0 else "P"

        poles, b, den = closed_loop_pole_b_pd_fo(Kp, Kd, tau)
        stable = np.all(np.real(poles) < 0)

        c_complex = closed_form_pd_fo_response(t, Kp, Kd, tau, b)
        c = np.real(c_complex)

        c_inf_from_poles = Kp / ((tau + Kd) * b)
        c_inf = float(np.real_if_close(c_inf_from_poles))

        def c_of_t(tau_):
            y = closed_form_pd_fo_response(
                np.array([tau_]), Kp, Kd, tau, b
            )
            return float(np.real(y[0]))

        pole_constants_name = "b = -p"

    # Rise time:
    # Same convention as the second-order notebook:
    # first time c(t) reaches the unit reference c = 1.
    tr = first_crossing_time(c_of_t, t, c, 1.0)

    # First crossing of 95% and 98% of final value
    t95 = first_crossing_time(c_of_t, t, c, 0.95*c_inf)
    t98 = first_crossing_time(c_of_t, t, c, 0.98*c_inf)

    # Settling time using 5% and 2% bands
    Ts_5 = settling_time(t, c, c_inf, 0.05)
    Ts_2 = settling_time(t, c, c_inf, 0.02)

    # Peak information
    peak_index = np.argmax(c)
    t_peak = t[peak_index]
    c_peak = c[peak_index]
    Mp = c_peak - c_inf

    if abs(c_inf) > 1e-12:
        percent_overshoot = 100.0 * Mp / c_inf
    else:
        percent_overshoot = np.nan

    # Plot
    plt.figure(figsize=(9, 5))
    plt.plot(t, c, linewidth=2,
             label=rf"$c(t)$ from derived {controller_type} equation")

    plt.axhline(1.0, linestyle="--", linewidth=1, label="unit reference")

    # Settling bands around c_inf
    plt.axhline(1.05*c_inf, linestyle="--", linewidth=1,
                label=r"$\pm 5\%$ band around $c(\infty)$")
    plt.axhline(0.95*c_inf, linestyle="--", linewidth=1)

    plt.axhline(1.02*c_inf, linestyle="-.", linewidth=1,
                label=r"$\pm 2\%$ band around $c(\infty)$")
    plt.axhline(0.98*c_inf, linestyle="-.", linewidth=1)

    # Rise time marker
    if tr is not None:
        plt.axvline(tr, linestyle="--", linewidth=1,
                    label=rf"$t_r={tr:.3g}$")
        plt.plot(tr, 1.0, "o")

    # 95% and 98% first-crossing markers
    if t95 is not None:
        plt.plot(t95, 0.95*c_inf, "o", label=rf"$t_{{95}}={t95:.3g}$")

    if t98 is not None:
        plt.plot(t98, 0.98*c_inf, "o", label=rf"$t_{{98}}={t98:.3g}$")

    # Peak marker
    plt.plot(t_peak, c_peak, "s", label=rf"peak at $t_p={t_peak:.3g}$")

    # Settling time markers
    if Ts_5 is not None:
        plt.axvline(Ts_5, linestyle="--", linewidth=1,
                    label=rf"$T_{{s,5\%}}={Ts_5:.3g}$")

    if Ts_2 is not None:
        plt.axvline(Ts_2, linestyle="-.", linewidth=1,
                    label=rf"$T_{{s,2\%}}={Ts_2:.3g}$")

    plt.xlabel("Time, t")
    plt.ylabel("Response, c(t)")
    plt.title(f"{controller_type}-Controlled First-Order Plant Using Derived Closed-Form Equation")
    plt.grid(True)
    plt.legend(loc="best")
    plt.show()

    # Print information
    print(f"Controller type: {controller_type}")
    print()

    if Ki > 0:
        print("Closed-loop characteristic polynomial:")
        print("(tau + Kd)*s^2 + (1 + Kp)*s + Ki")
        print(f"tau + Kd = {den[0]:.6g}")
        print(f"1 + Kp  = {den[1]:.6g}")
        print(f"Ki      = {den[2]:.6g}")
    else:
        print("Closed-loop characteristic polynomial:")
        print("(tau + Kd)*s + (1 + Kp)")
        print(f"tau + Kd = {den[0]:.6g}")
        print(f"1 + Kp  = {den[1]:.6g}")

    print()

    print("Closed-loop poles p_i:")
    for k, p in enumerate(poles, start=1):
        print(f"p{k} = {p:.6g}")
    print()

    print(f"Values used in the derived equation, {pole_constants_name}:")
    if Ki > 0:
        for k, ak in enumerate(a, start=1):
            print(f"a{k} = {ak:.6g}")
    else:
        print(f"b = {b:.6g}")

    print()
    print(f"Stable?      = {stable}")
    print(f"tau          = {tau:.6g}")
    print()

    if Ki > 0:
        print(f"Product a1*a2 = {np.prod(a):.6g}")
        print(f"Ki/(tau+Kd)  = {Ki/(tau+Kd):.6g}")
        print(f"c(infty) from Ki/((tau+Kd)*a1*a2) = {c_inf:.6g}")
        print("Note: PID/PI control gives c(infty)=1 if stable and Ki>0.")
    else:
        print(f"b             = {b:.6g}")
        print(f"c(infty) from Kp/((tau+Kd)*b) = {c_inf:.6g}")
        print("Equivalent: c(infty) = Kp/(1+Kp).")
        print("Note: PD/P control does not generally give c(infty)=1.")

    print()

    print(f"c(0+)        = {c[0]:.6g}")
    print(f"max c(t)     = {c_peak:.6g}")
    print(f"t_peak       = {t_peak:.6g}")
    print(f"Mp           = {Mp:.6g}")
    print(f"%OS          = {percent_overshoot:.6g} %")
    print()

    if tr is not None:
        print(f"rise time tr, c(t)=1        = {tr:.6g}")
    else:
        print("rise time tr, c(t)=1        = not found in plotted time range")

    if t95 is not None:
        print(f"t95 first crossing          = {t95:.6g}")
    else:
        print("t95 first crossing          = not found in plotted time range")

    if t98 is not None:
        print(f"t98 first crossing          = {t98:.6g}")
    else:
        print("t98 first crossing          = not found in plotted time range")

    if Ts_5 is not None:
        print(f"settling time 5%            = {Ts_5:.6g}")
    else:
        print("settling time 5%            = not settled in plotted time range")

    if Ts_2 is not None:
        print(f"settling time 2%            = {Ts_2:.6g}")
    else:
        print("settling time 2%            = not settled in plotted time range")
